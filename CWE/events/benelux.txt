country_event = {
  id = 7000003
  title = "EVT_7000003_NAME"
  desc = "EVT_7000003_DESC"
  picture = "Benelux_GeneralPicture"

  trigger = { 
    tag = BNL 
    ruling_party_ideology = nationalist
    primary_culture = dutch
  }

  mean_time_to_happen = { months = 2 }

  option = {
    name = "EVT_7000003_A"
    remove_accepted_culture = wallonian
    remove_accepted_culture = german

    government = populist_dictatorship
    ruling_party_ideology  = populist

    BNL = { any_pop = { limit = { OR = { culture = wallonian culture = german } } militancy = 5 consciousness = 5 } }
    ai_chance = { factor = 0 }
  }

  option = {
    name = "EVT_7000003_B"

    BNL = { all_core = { remove_core = BNL } }
    change_tag = NET
    remove_accepted_culture = flemish
    remove_accepted_culture = wallonian
    remove_accepted_culture = german
    release = BEL
    release = LUX
    ai_chance = { factor = 100 }
  }
}
